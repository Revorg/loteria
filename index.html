<script>

    const API = "https://script.google.com/macros/s/AKfycbxAFMnKjPhMBlPwOfThHbNh4F0fm55xA3Pm470Cu-YBJ3-iZnEvGhvQ14U1fvaKWMLIEg/exec";

    const WA_TLF = "584241868334";

    let SELECCIONADO = null;

    let ID_VALIDO = null;


    window.onload = load;


    function load() {

        fetch(API).then(r => r.json()).then(data => {

            const g = document.getElementById('grid');

            g.innerHTML = '';

            let l=0, a=0, p=0;

            data.forEach(i => {

                const d = document.createElement('div');

                const est = i.estado.toLowerCase().trim();

                d.innerText = i.n < 10 ? '0'+i.n : i.n;

                d.className = `numero ${est}`;

                if(est === 'libre') { d.onclick = () => openM(i.n); l++; }

                else if(est === 'pagado') p++; else a++;

                g.appendChild(d);

            });

            document.getElementById('l-c').innerText = l;

            document.getElementById('a-c').innerText = a;

            document.getElementById('p-c').innerText = p;

        });

    }


    function buscarID() {

        const id = document.getElementById('id-b').value.trim();

        if(!id) return alert("Pon el ID");

        fetch(`${API}?buscarID=${id}`).then(r => r.json()).then(res => {

            if(res.status === "Encontrado") {

                ID_VALIDO = id;

                document.getElementById('info-res').innerHTML = `✅ HALLADO: ${res.nombre}<br>Nro: ${res.numero}`;

                document.getElementById('pago-form').style.display = 'block';

            } else alert("❌ No hallado");

        });

    }


    function enviarPagoWA() {

    const b = document.getElementById('bnc').value;

    const r = document.getElementById('ref').value;

    if(!b || !r) return alert("Faltan datos del pago");


    // 🚀 Primero registra en la hoja

    fetch(API, {

        method: "POST",

        body: JSON.stringify({

            tipo: "PAGO",

            id: ID_VALIDO,

            banco: b,

            referencia: r,

            monto: 2500 // El monto de tu promoción

        })

    }).then(() => {

        // 📲 Luego abre WhatsApp

        const msg = encodeURIComponent(`¡Listo! Reporto pago:\nID: ${ID_VALIDO}\nBanco: ${b}\nRef: ${r}`);

        window.open(`https://wa.me/${WA_TLF}?text=${msg}`);

        alert("✅ Pago registrado y notificado.");

        location.reload();

    });

}


    function verTab(n) {

        document.getElementById('t1_cont').style.display = n===1 ? 'block' : 'none';

        document.getElementById('t2_cont').style.display = n===2 ? 'block' : 'none';

        document.getElementById('b1').className = `tab ${n===1?'active':''}`;

        document.getElementById('b2').className = `tab ${n===2?'active':''}`;

    }

    function openM(n) { SELECCIONADO = n; document.getElementById('num-span').innerText = n; document.getElementById('m-res').style.display = 'flex'; }

    function cerrarM() { document.getElementById('m-res').style.display = 'none'; }
